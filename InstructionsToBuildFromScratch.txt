Zero to Hero

Install OS
-Open Raspberry Pi Imager
-Install Raspberry Pi OS (32-bit)

Configure OS
-welcome to Raspberry Pi configuration
    -skip "Select Wireless Network"
    -skip "Update Software"
-sudo apt-get update
-sudo apt-get upgrade

Network Configuration
-sudo nano /etc/dhcpcd.conf
    '''
    #at the end of the file
    interface wlan0
        static ip_address=10.10.10.1/24
        nohook wpa_wupplicant
    '''
-sudo service dhcpcd restart

-sudo apt install dnsmasq
-sudo systemctl stop dnsmasq

-sudo mv /etc/dnsmasq.conf /etc/dnsmasq.conf.orig #discard old config file
-sudo nano /etc/dnsmasq.conf
    '''
    interface=wlan0
    dhcp-range=10.10.10.100,10.10.10.140,255.255.255.0,24h
        #DHCP pool which assigns ips for 24 hours
    domain=wlan #local wireless dns system
    address=/toybox.com/10.10.10.1 #reach Pi with toybox.com
    '''
-sudo systemctl start dnsmasq        

-sudo rfkill unblock wlan #wifi radio not blocked
-sudo apt install hostapd
-sudo systemctl stop hostapd
-sudo nano /etc/hostapd/hostapd.conf
    '''
    country_code=US
    interface=wlan0
    ssid=Toybox
    hw_mode=g
    channel=7
    wmm_enabled=0
    macaddr_acl=0
    auth_algs=1
    ignore_broadcast_ssid=0
    wpa=2
    wpa_passphrase=raspberry
    wpa_key_mgmt=WPA-PSK
    wpa_pairwise=TKIP
    rsn_pairwise=CCMP
    '''
-sudo nano /etc/default/hostapd
    '''
    #find line with "#DAEMON_CONF" and replace with
    DAEMON_CONF="/etc/hostapd/hostapd.conf"
    '''
-sudo systemctl unmask hostapd
-sudo systemctl enable hostapd
-sudo systemctl start hostapd

Setup Snake
-sudo apt install apache2
-sudo apt install nodejs npm

-wget https://github.com/FernandesBen/OnlineSnakeGame/archive/master.zip
-unzip master.zip -d master
-sudo chmod u+x master -R 
-npm install

-sudo nano OnlineSnakeGame.sh
    '''
    #!/bin/sh
    node /home/pi/master/OnlineSnakeGame-master/server.js
    '''
-sudo cp OnlineSnakeGame.sh /usr/bin/OnlineSnakeGame.sh
-sudo chmod +x /usr/bin/OnlineSnakeGame.sh
-sudo nano /etc/systemd/system/OnlineSnakeGame.service
    '''
    [Unit]
    Description=Starts OnlineSnakeGame

    [Service]
    Type=simple
    ExecStart=/bin/bash /usr/bin/OnlineSnakeGame.sh

    [Install]
    WantedBy=multi-user.target
    '''
-sudo chmod 644 /etc/systemd/system/OnlineSnakeGame.service
-sudo systemctl start OnlineSnakeGame
    -sudo systemctl status OnlineSnakeGame
-sudo systemctl enable OnlineSnakeGame
-sudo systemctl reboot
